An abstraction over the Hacker News JSON API. Given a short path, it expands it out and makes sure no requests are cached by the browser, then returns the result in a JSON format. hnFetch() also handles caching, so multiple requests about the same thing only result on one request using the CACHE, which is a map from API routes to responses.


16const CACHE = new Map();


17const hnFetch = async (apiPath, skipCache) => {


18    if (!CACHE.has(apiPath) || skipCache) {


19        const result = await fetch(API_ROOT + apiPath + '.json', {


20            cache: 'no-cache',


21        }).then(resp => resp.json());


22        CACHE.set(apiPath, result)


23        return result;


24    } else {


25        return CACHE.get(apiPath);


26    }


27}

A date formatter that does relative dates in English for the last 2 days.


37const formatDate = unix => {


38    if (!unix) return 'some time ago';


39


40    const date = new Date(unix * 1000);


41    const delta = (NOW - date) / 1000;


42    if (delta < 60) {


43        return '&#60;1 min ago';


44    } else if (delta < 3600) {


45        return `${~~(delta / 60)} min ago`;


46    } else if (delta < 86400) {


47        return `${~~(delta / 3600)} hr ago`;


48    } else if (delta < 86400 * 2) {


49        return 'yesterday';


50    } else if (delta < 86400 * 3) {


51        return '2 days ago';


52    } else {


53        return date.toLocaleDateString() + ' ' + formatTime(date) + ' ago';


54    }


55}
