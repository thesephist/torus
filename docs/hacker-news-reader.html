An abstraction over the Hacker News JSON API. Given a short path, it expands it out and makes sure no requests are cached by the browser, then returns the result in a JSON format. hnFetch() also handles caching, so multiple requests about the same thing only result on one request using the CACHE, which is a map from API routes to responses.


17const CACHE = new Map();


18const hnFetch = async (apiPath, skipCache) => {


19    if (!CACHE.has(apiPath) || skipCache) {


20        const result = await fetch(HN_API_ROOT + apiPath + '.json', {


21            cache: 'no-cache',


22        }).then(resp => resp.json());


23        CACHE.set(apiPath, result)


24        return result;


25    } else {


26        return CACHE.get(apiPath);


27    }


28}

A date formatter that does relative dates in English for the last 2 days.


50const formatDate = unix => {


51    if (!unix) return 'some time ago';


52


53    const date = new Date(unix * 1000);


54    const delta = (NOW - date) / 1000;


55    if (delta < 60) {


56        return '< 1 min ago';


57    } else if (delta < 3600) {


58        return `${~~(delta / 60)} min ago`;


59    } else if (delta < 86400) {


60        return `${~~(delta / 3600)} hr ago`;


61    } else if (delta < 86400 * 2) {


62        return 'yesterday';


63    } else if (delta < 86400 * 3) {


64        return '2 days ago';


65    } else {


66        return date.toLocaleDateString() + ' ' + formatTime(date) + ' ago';


67    }


68}
