An abstraction over the Hacker News JSON API. Given a short path, it expands it out and makes sure no requests are cached by the browser, then returns the result in a JSON format. hnFetch() also handles caching, so multiple requests about the same thing only result on one request using the CACHE, which is a map from API routes to responses.


22const CACHE = new Map();


23const hnFetch = async (apiPath, skipCache) => {


24    if (!CACHE.has(apiPath) || skipCache) {


25        const result = await fetch(HN_API_ROOT + apiPath + '.json', {


26            cache: 'no-cache',


27        }).then(resp => resp.json());


28        CACHE.set(apiPath, result)


29        return result;


30    } else {


31        return CACHE.get(apiPath);


32    }


33}

A date formatter that does relative dates in English for the last 2 days.


55const formatDate = unix => {


56    if (!unix) {


57        return 'some time ago';


58    }


59


60    const date = new Date(unix * 1000);


61    const delta = (NOW - date) / 1000;


62    if (delta < 60) {


63        return '< 1 min ago';


64    } else if (delta < 3600) {


65        return `${~~(delta / 60)} min ago`;


66    } else if (delta < 86400) {


67        return `${~~(delta / 3600)} hr ago`;


68    } else if (delta < 86400 * 2) {


69        return 'yesterday';


70    } else if (delta < 86400 * 3) {


71        return '2 days ago';


72    } else {


73        return date.toLocaleDateString() + ' ' + formatTime(date) + ' ago';


74    }


75}
